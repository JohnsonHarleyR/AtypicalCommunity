CREATE TABLE Entry (
    Id INTEGER PRIMARY KEY AUTOINCREMENT,
        UserId INT NOT NULL,
  'DateAndTime' DATETIME NOT NULL,
  `Happy` INT NOT NULL,
  `Sad` INT NOT NULL,
  `Confident` INT NOT NULL,
  `Mad` INT NOT NULL,
  `Hopeful` INT NOT NULL,
  `Scared` INT NOT NULL,
  `Title` VARCHAR(100) NOT NULL,
  `Text` BLOB NOT NULL,
  FOREIGN KEY (UserId) REFERENCES User(Id)
)
